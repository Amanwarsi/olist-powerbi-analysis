1) % Late Deliveries = DIVIDE(CALCULATE([Total Orders], Orders[is_late] = TRUE), [Total Orders])

2) % Low Reviews = DIVIDE(CALCULATE(COUNTROWS(Reviews), Reviews[review_score] <= 2), COUNTROWS(Reviews))

3) Avg Delivery Days = AVERAGE(Orders[delivery_days])

4) Avg freight per_order = DIVIDE([Total Freight], [Total Orders])

5) Avg Order Value = DIVIDE([Total Revenue], [Total Orders])

6) Avg Review Score = AVERAGE(Reviews[review_score])

7) Repeat Customers = 
COUNTROWS(
    FILTER(
        VALUES(Customers[customer_unique_id]),
        CALCULATE(DISTINCTCOUNT(Orders[order_id])) > 1
    )
)

8) Revenue YoY % = 
   VAR CurrentRev = [Total Revenue]
   VAR PriorRev = CALCULATE([Total Revenue], DATEADD('Date'[Date], -1, YEAR))
   RETURN DIVIDE(CurrentRev - PriorRev, PriorRev)

9) Total Freight = SUM('Order Items'[freight_value])

10) Total Orders = DISTINCTCOUNT(Orders[order_id])

11) Total Revenue = SUM('Order Items'[price])
